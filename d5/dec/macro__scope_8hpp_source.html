99#define NLOHMANN_JSON_SERIALIZE_ENUM(ENUM_TYPE, ...) \
100 template<typename BasicJsonType> \
101 inline void to_json(BasicJsonType& j, const ENUM_TYPE& e) \
102 { \
103 static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
104 static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
105 auto it = std::find_if(std::begin(m), std::end(m), \
106 [e](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
107 { \
108 return ej_pair.first == e; \
109 }); \
110 j = ((it != std::end(m)) ? it : std::begin(m))->second; \
111 } \
112 template<typename BasicJsonType> \
113 inline void from_json(const BasicJsonType& j, ENUM_TYPE& e) \
114 { \
115 static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
116 static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
117 auto it = std::find_if(std::begin(m), std::end(m), \
118 [&j](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
119 { \
120 return ej_pair.second == j; \
121 }); \
122 e = ((it != std::end(m)) ? it : std::begin(m))->first; \
123 }

303#define NLOHMANN_CAN_CALL_STD_FUNC_IMPL(std_name) \
304 namespace detail { \
305 using std::std_name; \
306 \
307 template<typename... T> \
308 using result_of_##std_name = decltype(std_name(std::declval<T>()...)); \
309 } \
310 \
311 namespace detail2 { \
312 struct std_name##_tag \
313 { \
314 }; \
315 \
316 template<typename... T> \
317 std_name##_tag std_name(T&&...); \
318 \
319 template<typename... T> \
320 using result_of_##std_name = decltype(std_name(std::declval<T>()...)); \
321 \
322 template<typename... T> \
323 struct would_call_std_##std_name \
324 { \
325 static constexpr auto const value = ::nlohmann::detail:: \
326 is_detected_exact<std_name##_tag, result_of_##std_name, T...>::value; \
327 }; \
328 } /* namespace detail2 */ \

1105 void destroy(value_t t)
1106 {
1107 if (t == value_t::array || t == value_t::object)
1108 {
1109 // flatten the current json_value to a heap-allocated stack
1110 std::vector<basic_json> stack;
1111
1112 // move the top-level items to stack
1113 if (t == value_t::array)
1114 {
1115 stack.reserve(array->size());
1116 std::move(array->begin(), array->end(), std::back_inserter(stack));
1117 }
1118 else
1119 {
1120 stack.reserve(object->size());
1121 for (auto&& it : *object)
1122 {
1123 stack.push_back(std::move(it.second));
1124 }
1125 }
1126
1127 while (!stack.empty())
1128 {
1129 // move the last item to local variable to be processed
1130 basic_json current_item(std::move(stack.back()));
1131 stack.pop_back();
1132
1133 // if current_item is array/object, move
1134 // its children to the stack to be processed later
1135 if (current_item.is_array())
1136 {
1137 std::move(current_item.m_value.array->begin(), current_item.m_value.array->end(), std::back_inserter(stack));
1138
1139 current_item.m_value.array->clear();
1140 }
1141 else if (current_item.is_object())
1142 {
1143 for (auto&& it : *current_item.m_value.object)
1144 {
1145 stack.push_back(std::move(it.second));
1146 }
1147
1148 current_item.m_value.object->clear();
1149 }
1150
1151 // it's now safe that current_item get destructed
1152 // since it doesn't have any children
1153 }
1154 }
1155
1156 switch (t)
1157 {
1158 case value_t::object:
1159 {
1160 AllocatorType<object_t> alloc;
1161 std::allocator_traits<decltype(alloc)>::destroy(alloc, object);
1162 std::allocator_traits<decltype(alloc)>::deallocate(alloc, object, 1);
1163 break;
1164 }
1165
1166 case value_t::array:
1167 {
1168 AllocatorType<array_t> alloc;
1169 std::allocator_traits<decltype(alloc)>::destroy(alloc, array);
1170 std::allocator_traits<decltype(alloc)>::deallocate(alloc, array, 1);
1171 break;
1172 }
1173
1174 case value_t::string:
1175 {
1176 AllocatorType<string_t> alloc;
1177 std::allocator_traits<decltype(alloc)>::destroy(alloc, string);
1178 std::allocator_traits<decltype(alloc)>::deallocate(alloc, string, 1);
1179 break;
1180 }
1181
1182 case value_t::binary:
1183 {
1184 AllocatorType<binary_t> alloc;
1185 std::allocator_traits<decltype(alloc)>::destroy(alloc, binary);
1186 std::allocator_traits<decltype(alloc)>::deallocate(alloc, binary, 1);
1187 break;
1188 }

1190 case value_t::null:
1191 case value_t::boolean:
1192 case value_t::number_integer:
1193 case value_t::number_unsigned:
1194 case value_t::number_float:

8409 const auto get_op = [](const std::string & op)
8410 {
8411 if (op == "add")
8412 {
8413 return patch_operations::add;
8414 }
8415 if (op == "remove")
8416 {
8417 return patch_operations::remove;
8418 }

8420 {

8706 JSON_HEDLEY_WARN_UNUSED_RESULT
8707 static basic_json diff(const basic_json& source, const basic_json& target,
8708 const std::string& path = "")
8709 {
8710 // the patch
8711 basic_json result(value_t::array);
8712
8713 // if the values are the same, return empty patch
8714 if (source == target)
8715 {
8716 return result;
8717 }
8718
8719 if (source.type() != target.type())
8720 {
8721 // different types: replace value
8722 result.push_back(
8723 {
8724 {"op", "replace"}, {"path", path}, {"value", target}
8725 });
8726 return result;
8727 }
8728
8729 switch (source.type())
8730 {
8731 case value_t::array:
8732 {
8733 // first pass: traverse common elements
8734 std::size_t i = 0;
8735 while (i < source.size() && i < target.size())
8736 {
8737 // recursive call to compare array values at index i
8738 auto temp_diff = diff(source[i], target[i], path + "/" + std::to_string(i));
8739 result.insert(result.end(), temp_diff.begin(), temp_diff.end());
8740 ++i;
8741 }
8742
8743 // i now reached the end of at least one array
8744 // in a second pass, traverse the remaining elements
8745
8746 // remove my remaining elements
8747 const auto end_index = static_cast<difference_type>(result.size());
8748 while (i < source.size())
8749 {
8750 // add operations in reverse order to avoid invalid
8751 // indices
8752 result.insert(result.begin() + end_index, object(
8753 {
8754 {"op", "remove"},
8755 {"path", path + "/" + std::to_string(i)}
8756 }));
8757 ++i;
8758 }
8759
8760 // add other remaining elements
8761 while (i < target.size())
8762 {
8763 result.push_back(
8764 {
8765 {"op", "add"},
8766 {"path", path + "/-"},
8767 {"value", target[i]}
8768 });
8769 ++i;
8770 }
8771
8772 break;
8773 }
8774
8775 case value_t::object:
8776 {
8777 // first pass: traverse this object's elements
8778 for (auto it = source.cbegin(); it != source.cend(); ++it)
8779 {
8780 // escape the key name to be used in a JSON patch
8781 const auto path_key = path + "/" + detail::escape(it.key());
8782
8783 if (target.find(it.key()) != target.end())
8784 {
8785 // recursive call to compare object values at key it
8786 auto temp_diff = diff(it.value(), target[it.key()], path_key);
8787 result.insert(result.end(), temp_diff.begin(), temp_diff.end());
8788 }
8789 else
8790 {
8791 // found a key that is not in o -> remove it
8792 result.push_back(object(
8793 {
8794 {"op", "remove"}, {"path", path_key}
8795 }));
8796 }
8797 }
8798
8799 // second pass: traverse other object's elements
8800 for (auto it = target.cbegin(); it != target.cend(); ++it)
8801 {
8802 if (source.find(it.key()) == source.end())
8803 {
8804 // found a key that is not in this -> add it
8805 const auto path_key = path + "/" + detail::escape(it.key());
8806 result.push_back(
8807 {
8808 {"op", "add"}, {"path", path_key},
8809 {"value", it.value()}
8810 });
8811 }
8812 }
8813
8814 break;
8815 }
8816
8817 case value_t::null:
8818 case value_t::string:
8819 case value_t::boolean:
8820 case value_t::number_integer:
8821 case value_t::number_unsigned:
8822 case value_t::number_float:
8823 case value_t::binary:
8824 case value_t::discarded:
8825 default:
8826 {
8827 // both primitive type: replace value
8828 result.push_back(
8829 {
8830 {"op", "replace"}, {"path", path}, {"value", target}
8831 });
8832 break;
8833 }
8834 }
8835
8836 return result;
8837 }
